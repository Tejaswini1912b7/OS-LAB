AIM  :  To write a C program for implementation of Priority scheduling algorithms

ALGORITHM: 
START
	Declare and read the values of Burst Times and their Priorities for required number of processes.
	Based on the priority Time, Sort all the Processes.
 	Calculate the following:
	    1. Finishing Time = Burst Time + Waiting Time
	      (or) Waiting Time = Turnaround Time - Burst Time
	    2. Average Waiting Time and Average Turnaround Time.
STOP 

CODE:
// Priority Scheduling Algorithm
#include<stdio.h>
int main()
{
    int bt[20], p[20], wt[20], ft[20], pr[20];
    int i,j,n,total = 0,pos,temp;
    float avg_wt,avg_ft;
    printf("Enter Total Number of Processes : ");
    scanf("%d",&n);
    printf("\nEnter Time required for the Process and it's Priority\n");
    for(i = 0; i < n; i++)
    {
        printf("P[%d]\n",i+1);
        printf("Process Time :  ");
        scanf("%d",&bt[i]);
        printf("Priority     :  ");
        scanf("%d",&pr[i]);
        p[i] = i + 1;       
    }
    for(i = 0; i < n; i++)
    {
        pos = i;
        for(j = i + 1; j < n; j++)
        {
            if(pr[j] < pr[pos])
                pos = j;
        }
        temp = pr[i];
        pr[i] = pr[pos];
        pr[pos] = temp;

        temp = bt[i];
        bt[i] = bt[pos];
        bt[pos] = temp;

        temp = p[i];
        p[i] = p[pos];
        p[pos] = temp;
    }
    wt[0] = 0;
    for(i = 1; i < n; i++)
    {
        wt[i] = 0;
        for(j = 0;j < i; j++)
            wt[i] += bt[j];
        total += wt[i];
    }
    avg_wt = total / n;      
    total = 0;
    printf("\nBased on the Priority Order\n");
    printf("\nProcess[pid]\tBurst Time\t Waiting Time\t   Turn Arround Time");
    for(i = 0; i < n; i++)
    {
        ft[i] = bt[i] + wt[i];     
        total += ft[i];
        printf("\nP[%d]\t\t  %d\t\t    %d\t\t\t%d", p[i], bt[i], wt[i], ft[i]);
    }
    avg_ft = total/n; 
    printf("\n\nAverage Waiting Time   :  %f", avg_wt);
    printf("\nAverage TurnAround Time  :  %f", avg_ft);
    return 0;
}
